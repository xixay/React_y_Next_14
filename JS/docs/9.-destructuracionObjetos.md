## objetos
```js
const mascota = {
  nombre: 'Tom',
  edad: 10,
  vivo: true,
  razas: ['peludo', 'negro'],
};

console.log(mascota.nombre);
const nombreMascota = mascota.nombre;
console.log(nombreMascota);//{ nombre: 'Tom', edad: 10, vivo: true, razas: [ 'peludo', 'negro' ] }
```
## destructuracion
```js
const { nombre, edad, vivo, razas } = mascota;
console.log(nombre);//Tom
console.log(edad);//10
```
