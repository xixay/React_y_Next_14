## Recorrer elementos con map
- Si no se usa el map, se puede usar el ul y li, para listar
```jsx
import imgUno from './assets/images/img-1.jpg'

const MyButton = () => {
  return (
    <button>Soy un boton</button>
  )
 }

const OnlineText = () =>{
  return (
    <div>
      <h3>Online</h3>
    </div>
  )
}
const OfflineText = () =>{
  return (
    <div>
      <h3>Offline</h3>
    </div>
  )
}
const App = () => {

  const title = 'Mi titulo desde una constante';
  const classTitle = 'text-center';
  // const pathImg = 'src/assets/images/img-1.jpg';
  const user = false;

  const fruts =['manzana', 'platano', 'pera'];

  return (
    <>
      <h1 className={classTitle}>{title}</h1>
      <img src={imgUno} alt={`imagen-${title}`} />
      <MyButton></MyButton>
      {user && <OnlineText/>}
      <ul>
        <li>{fruts[0]}</li>
        <li>{fruts[1]}</li>
        <li>{fruts[2]}</li>
      </ul>
    </>
  )
};

export default App;
```

- Para solucionarlo le indicamos cual se renderizara primero, para ello le pasamos el props el index, y desaparecera el error
```jsx
import imgUno from './assets/images/img-1.jpg'

const MyButton = () => {
  return (
    <button>Soy un boton</button>
  )
 }

const OnlineText = () =>{
  return (
    <div>
      <h3>Online</h3>
    </div>
  )
}
const OfflineText = () =>{
  return (
    <div>
      <h3>Offline</h3>
    </div>
  )
}
const App = () => {

  const title = 'Mi titulo desde una constante';
  const classTitle = 'text-center';
  // const pathImg = 'src/assets/images/img-1.jpg';
  const user = false;

  const fruts =['manzana', 'platano', 'pera'];

  return (
    <>
      <h1 className={classTitle}>{title}</h1>
      <img src={imgUno} alt={`imagen-${title}`} />
      <MyButton></MyButton>
      {user && <OnlineText/>}
      <ul>
        {
          fruts.map((frut, index) => (
            <li key={index}>{frut} - {index}</li>
          ))
        }
      </ul>
    </>
  )
};

export default App;
```
