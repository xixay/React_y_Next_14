## Operador '&&'
- Se va omitir el else y en su lugar se colocara en lugar de un if, un operador &&
- Osea cuando user sea true, va mostrar el componente que lo acompaña
```jsx
const MyButton = () => {
  return (
    <button>Soy un boton</button>
  )
 }

const OnlineText = () =>{
  return (
    <div>
      <h3>Online</h3>
    </div>
  )
}
const OfflineText = () =>{
  return (
    <div>
      <h3>Offline</h3>
    </div>
  )
}
const App = () => {

  const title = 'Mi titulo desde una constante';
  const classTitle = 'text-center';
  const pathImg = 'https://picsum.photos/200/300';
  const user = true;

  return (
    <>
      <h1 className={classTitle}>{title}</h1>
      <img src={pathImg} alt={`imagen-${title}`} />
      <MyButton></MyButton>
      {/* pregunta si usuario es verdadero */}
      {
        user && <OnlineText/>
      }
    </>
  )
};

export default App;
```
