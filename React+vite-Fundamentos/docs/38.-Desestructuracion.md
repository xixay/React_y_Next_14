## Desestructuracion
- Si sabemos que props, es un objeto podemos sacar sus propiedades e inicializarlas, al hacerlo evitamos estar colocando props a cada rato
```jsx
import imgUno from './assets/images/img-1.jpg'

const MyButton = ({text}) => {
  // console.log(props)
  return (
    <button>{text}</button>
  )
 }

const OnlineText = () =>{
  return (
    <div>
      <h3>Online</h3>
    </div>
  )
}
const OfflineText = () =>{
  return (
    <div>
      <h3>Offline</h3>
    </div>
  )
}
const App = () => {

  const title = 'Mi titulo desde una constante';
  const classTitle = 'text-center';
  // const pathImg = 'src/assets/images/img-1.jpg';
  const user = false;

  const fruts =['manzana', 'platano', 'pera'];

  return (
    <>
      <h1 className={classTitle}>{title}</h1>
      <img src={imgUno} alt={`imagen-${title}`} />

      <MyButton text = 'botòn 1'/>
      <MyButton text = 'botòn 2'/>
      <MyButton text = 'botòn 3'/>
      {user && <OnlineText/>}
      <ul>
        {
          fruts.map((frut, index) => (
            <li key={index}>{frut} - {index}</li>
          ))
        }
      </ul>
    </>
  )
};

export default App;
```
