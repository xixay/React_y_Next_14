## Reutilizar componentes
- Se puede reutilizar varias veces el componente
```jsx
const MyButton = () => {
  return (
    <button>Soy un boton</button>
  )
 }

const OnlineText = () =>{
  return (
    <div>
      <h3>Online</h3>
    </div>
  )
}
const OfflineText = () =>{
  return (
    <div>
      <h3>Offline</h3>
    </div>
  )
}
const App = () => {

  const title = 'Mi titulo desde una constante';
  const classTitle = 'text-center';
  const pathImg = 'https://picsum.photos/200/300';
  const user = false;

  return (
    <>
      <h1 className={classTitle}>{title}</h1>
      <img src={pathImg} alt={`imagen-${title}`} />
      <MyButton></MyButton>
      <MyButton></MyButton>
      <MyButton></MyButton>
      <MyButton></MyButton>
      <MyButton></MyButton>
      <MyButton></MyButton>
      <MyButton></MyButton>
      <MyButton></MyButton>
      <MyButton></MyButton>
      <MyButton></MyButton>
      {user && <OnlineText/>}
    </>
  )
};

export default App;
```
