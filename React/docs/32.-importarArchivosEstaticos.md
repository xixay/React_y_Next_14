## Importar Archivos Estáticos
- crear la carpeta images, dentro de assets, y guardar una imagen, que sera usado en el pathImg, ya no sera un servidor externo
```jsx
const MyButton = () => {
  return (
    <button>Soy un boton</button>
  )
 }

const OnlineText = () =>{
  return (
    <div>
      <h3>Online</h3>
    </div>
  )
}
const OfflineText = () =>{
  return (
    <div>
      <h3>Offline</h3>
    </div>
  )
}
const App = () => {

  const title = 'Mi titulo desde una constante';
  const classTitle = 'text-center';
  const pathImg = './assets/images/img-1.jpg';
  const user = false;

  return (
    <>
      <h1 className={classTitle}>{title}</h1>
      <img src={pathImg} alt={`imagen-${title}`} />
      <MyButton></MyButton>
      {user && <OnlineText/>}
    </>
  )
};

export default App;
```

 }
