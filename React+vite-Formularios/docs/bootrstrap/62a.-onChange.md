## onChange
- En el archivo Controlado.jsx, colocar el onChange
```jsx
import { useState } from "react";

const Controlado = () => {
  //Variables de entrada
  const [title, setTitle] = useState('')
  const [description, setDescription] = useState('')
  const [state, setState] = useState('pendiente')
  //Se crea la funcion que capturara el evento
  const handleSubmit = (e) => {
    e.preventDefault();//Se coloca para que no envie por defecto al url

  };

  return (
    <form onSubmit={handleSubmit}>
      <input
        className="form-control mb-2"
        type="text"
        placeholder="Ingrese Todo"
        name="title"
        value={title}
        onChange={e => console.log(e.target.value)}
      />
      <textarea
        className="form-control mb-2"
        placeholder="Ingrese descripción"
        name="description"
        value={description}
        onChange={e => console.log(e.target.value)}
      />
      <select
        className="form-select mb-2"
        name="state"
        value={state}
        onChange={e => console.log(e.target.value)}
      >
        <option value="pendiente">Pendiente</option>
        <option value="completado">Completado</option>
      </select>
      <button className="btn btn-primary" type="submit">
        Procesar
      </button>
    </form>
  );
};
export default Controlado;

```

- Para poner datos por default, colocar en su useState, y luego imprimir por terminal
```jsx
import { useState } from "react";

const Controlado = () => {
  //Variables de entrada
  const [title, setTitle] = useState('Todo #01')
  const [description, setDescription] = useState('Descripción #01')
  const [state, setState] = useState('pendiente')
  //Se crea la funcion que capturara el evento
  const handleSubmit = (e) => {
    e.preventDefault();//Se coloca para que no envie por defecto al url
    //Enviar los Datos
    console.log(title);
    console.log(description);
    console.log(state);
  };

  return (
    <form onSubmit={handleSubmit}>
      <input
        className="form-control mb-2"
        type="text"
        placeholder="Ingrese Todo"
        name="title"
        value={title}
        onChange={e => setTitle(e.target.value)}
      />
      <textarea
        className="form-control mb-2"
        placeholder="Ingrese descripción"
        name="description"
        value={description}
        onChange={e => setDescription(e.target.value)}
      />
      <select
        className="form-select mb-2"
        name="state"
        value={state}
        onChange={e => setState(e.target.value)}
      >
        <option value="pendiente">Pendiente</option>
        <option value="completado">Completado</option>
      </select>
      <button className="btn btn-primary" type="submit">
        Procesar
      </button>
    </form>
  );
};
export default Controlado;
```
