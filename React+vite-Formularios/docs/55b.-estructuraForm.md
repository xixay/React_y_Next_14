## Estructura Formulario JSX
- Instalar PrimeFlex
```console
npm install primeflex
```
- Sabiendo que App.jsx era de la forma:
```jsx
import Nocontrolado from './components/NoControlado';

const App = () => {
  return (
    <div className='container'>
      <h1>Formularios</h1>
      <Nocontrolado />
    </div>
  );
};
export default App;
```
- En el archivo NoControlado.jsx, donde name nos servira para el form data, en ves de usar id, para cada uno
```jsx
import { InputText } from 'primereact/inputtext';
import { InputTextarea } from 'primereact/inputtextarea';
import { Dropdown } from 'primereact/dropdown';
import { Button } from 'primereact/button';

const Nocontrolado = () => {
  const options = [
    { label: 'Pendiente', value: 'pendiente' },
    { label: 'Completado', value: 'completado' },
  ];

  return (
    <form className="p-fluid">
      <div className="field mb-3">
        <label htmlFor="title" className="block mb-2">
          Ingrese Todo
        </label>
        <InputText id="title" name="title" />
      </div>

      <div className="field mb-3">
        <label htmlFor="description" className="block mb-2">
          Ingrese descripción
        </label>
        <InputTextarea id="description" name="description" rows={5} />
      </div>

      <div className="field mb-3">
        <label htmlFor="state" className="block mb-2">
          Seleccione un estado
        </label>
        <Dropdown id="state" name="state" options={options} placeholder="Seleccione un estado" />
      </div>

      <Button label="Procesar" icon="pi pi-check" type="submit" className="w-full" />
    </form>
  );
};
export default Nocontrolado;

```
